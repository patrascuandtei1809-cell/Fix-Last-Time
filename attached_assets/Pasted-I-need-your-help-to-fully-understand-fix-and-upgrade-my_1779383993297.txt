"""
bot.py - Core trading bot engine
Handles: data fetching, bot loop, manual trades, logging
"""

import time
import threading
from datetime import datetime
import pandas as pd
import requests
import numpy as np

from strategy import generate_signal
from risk import RiskManager


class TradingBot:
    """
    Singleton-style bot. One instance lives in st.session_state['bot'].
    Never re-instantiated on Streamlit rerun.
    """

    def __init__(self):
        # ── State ──────────────────────────────────────────────
        self.running = False
        self.auto_mode = False
        self._thread = None
        self._lock = threading.Lock()

        # ── Account ────────────────────────────────────────────
        self.balance = 1000.0
        self.initial_balance = 1000.0
        self.position = None          # dict or None
        self.trades = []              # list of closed trade dicts
        self.logs = []                # activity log entries
        self.max_logs = 100

        # ── Market data ────────────────────────────────────────
        self.df = pd.DataFrame()      # OHLC + indicators
        self.last_price = None
        self.last_fetch_ts = None
        self.fetch_error = None

        # ── Risk ───────────────────────────────────────────────
        self.risk = RiskManager(
            balance=self.balance,
            risk_pct=0.02,           # risk 2% of balance per trade
            stop_loss_pct=0.015,     # 1.5% stop loss
            take_profit_pct=0.03     # 3% take profit
        )

    # ──────────────────────────────────────────────────────────
    # LOGGING
    # ──────────────────────────────────────────────────────────

    def log(self, message: str, level: str = "INFO"):
        """Thread-safe log append."""
        ts = datetime.now().strftime("%H:%M:%S")
        entry = {
            "time": ts,
            "level": level,        # INFO | TRADE | ERROR | SYSTEM | MANUAL
            "message": message
        }
        with self._lock:
            self.logs.append(entry)
            if len(self.logs) > self.max_logs:
                self.logs.pop(0)

    # ──────────────────────────────────────────────────────────
    # DATA FETCHING (Binance Public API — no key needed)
    # ──────────────────────────────────────────────────────────

    def fetch_ohlc(self, symbol: str = "BTCUSDT",
                   interval: str = "5m",
                   limit: int = 200) -> pd.DataFrame:
        """
        Fetch OHLC candles from Binance.
        Returns empty DataFrame on failure (never returns fake data).
        """
        url = "https://api.binance.com/api/v3/klines"
        params = {"symbol": symbol, "interval": interval, "limit": limit}

        try:
            resp = requests.get(url, params=params, timeout=10)
            resp.raise_for_status()
            raw = resp.json()

            df = pd.DataFrame(raw, columns=[
                "open_time", "open", "high", "low", "close", "volume",
                "close_time", "quote_vol", "trades",
                "taker_buy_base", "taker_buy_quote", "ignore"
            ])

            df["timestamp"] = pd.to_datetime(df["open_time"], unit="ms")
            for col in ["open", "high", "low", "close", "volume"]:
                df[col] = df[col].astype(float)

            df = df[["timestamp", "open", "high", "low", "close", "volume"]].copy()
            df.reset_index(drop=True, inplace=True)

            self.fetch_error = None
            return df

        except requests.exceptions.ConnectionError:
            self.fetch_error = "Connection error — check internet"
        except requests.exceptions.Timeout:
            self.fetch_error = "Binance API timeout"
        except requests.exceptions.HTTPError as e:
            self.fetch_error = f"HTTP error: {e}"
        except Exception as e:
            self.fetch_error = f"Unexpected error: {e}"

        self.log(f"Fetch failed: {self.fetch_error}", "ERROR")
        return pd.DataFrame()

    def fetch_price(self, symbol: str = "BTCUSDT") -> float | None:
        """Fetch single latest price (fast ticker endpoint)."""
        url = "https://api.binance.com/api/v3/ticker/price"
        try:
            resp = requests.get(url, params={"symbol": symbol}, timeout=5)
            resp.raise_for_status()
            return float(resp.json()["price"])
        except Exception as e:
            self.log(f"Price fetch failed: {e}", "ERROR")
            return None

    # ──────────────────────────────────────────────────────────
    # POSITION MANAGEMENT
    # ──────────────────────────────────────────────────────────

    def _open_position(self, price: float, timestamp, reason: str):
        """Open a BUY position. Only one position at a time."""
        if self.position is not None:
            return  # silent — already in position

        size = self.risk.position_size(price)
        tp = round(price * (1 + self.risk.take_profit_pct), 2)
        sl = round(price * (1 - self.risk.stop_loss_pct), 2)

        self.position = {
            "side": "BUY",
            "entry_price": price,
            "entry_time": timestamp,
            "size": size,
            "take_profit": tp,
            "stop_loss": sl,
            "reason": reason
        }

        self.log(
            f"OPEN BUY @ ${price:,.2f} | "
            f"Size: {size:.6f} BTC | "
            f"TP: ${tp:,.2f} | SL: ${sl:,.2f} | "
            f"Reason: {reason}",
            "TRADE"
        )

    def _close_position(self, price: float, timestamp, reason: str):
        """Close the current position and record the trade."""
        if self.position is None:
            return

        entry = self.position["entry_price"]
        size = self.position["size"]
        pnl = round((price - entry) * size, 4)
        self.balance = round(self.balance + pnl, 4)

        # Update risk manager with new balance
        self.risk.balance = self.balance

        trade = {
            "entry_time": self.position["entry_time"],
            "exit_time": timestamp,
            "entry_price": entry,
            "exit_price": price,
            "size": size,
            "pnl": pnl,
            "reason": reason,
            "side": "BUY"
        }
        self.trades.append(trade)

        self.log(
            f"CLOSE @ ${price:,.2f} | "
            f"PnL: {'+'if pnl>=0 else ''}{pnl:.4f} USDT | "
            f"Balance: ${self.balance:,.2f} | "
            f"Reason: {reason}",
            "TRADE"
        )
        self.position = None

    def _check_tp_sl(self, price: float, timestamp):
        """Check take profit / stop loss against current price."""
        if self.position is None:
            return

        if price >= self.position["take_profit"]:
            self._close_position(price, timestamp, "Take Profit hit ✅")
        elif price <= self.position["stop_loss"]:
            self._close_position(price, timestamp, "Stop Loss hit 🛑")

    # ──────────────────────────────────────────────────────────
    # MANUAL TRADING
    # ──────────────────────────────────────────────────────────

    def manual_buy(self):
        """Execute manual BUY using latest price."""
        if self.df.empty:
            self.log("No data available for manual BUY", "ERROR")
            return

        price = float(self.df["close"].iloc[-1])
        ts = self.df["timestamp"].iloc[-1]

        if self.position is not None:
            self.log(
                f"Manual BUY rejected — already in position "
                f"(entry: ${self.position['entry_price']:,.2f})",
                "INFO"
            )
            return

        self._open_position(price, ts, "Manual BUY 🖱️")

    def manual_sell(self):
        """Execute manual SELL (close current position)."""
        if self.df.empty:
            self.log("No data available for manual SELL", "ERROR")
            return

        price = float(self.df["close"].iloc[-1])
        ts = self.df["timestamp"].iloc[-1]

        if self.position is None:
            self.log("Manual SELL rejected — no open position", "INFO")
            return

        self._close_position(price, ts, "Manual SELL 🖱️")

    # ──────────────────────────────────────────────────────────
    # BOT LOOP
    # ──────────────────────────────────────────────────────────

    def _bot_loop(self):
        """
        Main background loop.
        Runs every 60s. Only trades in auto_mode.
        Data is always refreshed so the chart stays live.
        """
        self.log("Bot loop started", "SYSTEM")

        while self.running:
            try:
                # 1. Fetch fresh OHLC
                df = self.fetch_ohlc()

                if df.empty:
                    # No fake fallback — just wait and retry
                    self.log("Waiting for valid data...", "INFO")
                    time.sleep(30)
                    continue

                # 2. Add indicators
                df = self._add_indicators(df)

                # 3. Get latest price (fast endpoint)
                live_price = self.fetch_price()
                if live_price:
                    self.last_price = live_price

                # 4. Store enriched dataframe
                with self._lock:
                    self.df = df.copy()
                    self.last_fetch_ts = datetime.now()

                # 5. Check TP/SL on live price
                if self.position and live_price:
                    self._check_tp_sl(
                        live_price,
                        datetime.now()
                    )

                # 6. Auto trading logic (only if auto_mode is ON)
                if self.auto_mode:
                    signal = generate_signal(df)
                    self._execute_signal(signal, df)

            except Exception as e:
                self.log(f"Bot loop error: {e}", "ERROR")

            # Wait 60 seconds between iterations
            # Check running flag every second so stop() is responsive
            for _ in range(60):
                if not self.running:
                    break
                time.sleep(1)

        self.log("Bot loop stopped", "SYSTEM")

    def _add_indicators(self, df: pd.DataFrame) -> pd.DataFrame:
        """Add EMA, RSI to dataframe."""
        close = df["close"]

        # EMA
        df["ema9"] = close.ewm(span=9, adjust=False).mean()
        df["ema21"] = close.ewm(span=21, adjust=False).mean()
        df["ema50"] = close.ewm(span=50, adjust=False).mean()

        # RSI
        df["rsi"] = self._calc_rsi(close, 14)

        return df

    @staticmethod
    def _calc_rsi(series: pd.Series, period: int = 14) -> pd.Series:
        delta = series.diff()
        gain = delta.clip(lower=0)
        loss = -delta.clip(upper=0)
        avg_gain = gain.ewm(com=period - 1, min_periods=period).mean()
        avg_loss = loss.ewm(com=period - 1, min_periods=period).mean()
        rs = avg_gain / avg_loss.replace(0, np.nan)
        return 100 - (100 / (1 + rs))

    def _execute_signal(self, signal: dict, df: pd.DataFrame):
        """Execute auto trade signal — clean single-position logic."""
        action = signal.get("action")     # "BUY" | "SELL" | "HOLD"
        reason = signal.get("reason", "")
        latest = df.iloc[-1]
        price = latest["close"]
        ts = latest["timestamp"]

        if action == "BUY" and self.position is None:
            self._open_position(price, ts, f"Auto: {reason}")

        elif action == "SELL" and self.position is not None:
            self._close_position(price, ts, f"Auto: {reason}")

        # HOLD or mismatched state → do nothing, no log spam

    # ──────────────────────────────────────────────────────────
    # START / STOP
    # ──────────────────────────────────────────────────────────

    def start(self):
        """Start bot. Safe to call multiple times — won't spawn extra threads."""
        if self.running:
            self.log("Bot already running", "INFO")
            return

        # Extra safety: check thread is not alive
        if self._thread and self._thread.is_alive():
            self.log("Thread already alive — skipping start", "INFO")
            return

        self.running = True
        self._thread = threading.Thread(
            target=self._bot_loop,
            daemon=True,
            name="TradingBotThread"
        )
        self._thread.start()
        self.log("Bot started ▶", "SYSTEM")

    def stop(self):
        """Gracefully stop the bot loop."""
        if not self.running:
            self.log("Bot already stopped", "INFO")
            return
        self.running = False
        self.log("Bot stopping... ⏹", "SYSTEM")

    def toggle_auto(self):
        """Toggle auto trading mode."""
        self.auto_mode = not self.auto_mode
        state = "ON 🤖" if self.auto_mode else "OFF 🖐️"
        self.log(f"Auto trading: {state}", "SYSTEM")

    # ──────────────────────────────────────────────────────────
    # STATS (for dashboard display)
    # ──────────────────────────────────────────────────────────

    def get_stats(self) -> dict:
        """Return summary stats for display."""
        total_trades = len(self.trades)

            "win_rate": win_rate,
            "in_position": self.position is not None,
            "position": self.position
        }

"""
strategy.py - Trading signal generation
Clean, readable, no side effects.
Input: enriched DataFrame (with indicators)
Output: {"action": "BUY"|"SELL"|"HOLD", "reason": str}
"""

import pandas as pd


def generate_signal(df: pd.DataFrame) -> dict:
    """
    EMA crossover + RSI filter strategy.

    BUY  when: EMA9 crosses above EMA21 AND RSI < 65 (not overbought)
    SELL when: EMA9 crosses below EMA21 AND RSI > 35 (not oversold)
    HOLD otherwise.
    """

    if len(df) < 3:
        return {"action": "HOLD", "reason": "Not enough data"}

    # Use last 2 rows to detect crossover
    prev = df.iloc[-2]
    curr = df.iloc[-1]

    # Guard: check required columns exist
    required = ["ema9", "ema21", "rsi"]
    for col in required:
        if col not in df.columns or pd.isna(curr[col]):
            return {"action": "HOLD", "reason": f"Indicator '{col}' not ready"}

    ema9_crossed_up = (prev["ema9"] < prev["ema21"]) and (curr["ema9"] > curr["ema21"])
    ema9_crossed_down = (prev["ema9"] > prev["ema21"]) and (curr["ema9"] < curr["ema21"])

    rsi = curr["rsi"]
    rsi_not_overbought = rsi < 65
    rsi_not_oversold = rsi > 35

    # ── BUY signal ──────────────────────────────────────────
    if ema9_crossed_up and rsi_not_overbought:
        return {
            "action": "BUY",
            "reason": f"EMA9 crossed above EMA21 | RSI={rsi:.1f}"
        }

    elif price < ema50 * 0.995:

"""
risk.py - Risk management
Handles: position sizing, daily loss limits
"""


class RiskManager:
    def __init__(
        self,
        balance: float,
        risk_pct: float = 0.02,
        stop_loss_pct: float = 0.015,
        take_profit_pct: float = 0.03
    ):
        self.balance = balance
        self.risk_pct = risk_pct
        self.stop_loss_pct = stop_loss_pct
        self.take_profit_pct = take_profit_pct

    def position_size(self, entry_price: float) -> float:
        """
        Calculate position size in BTC.

        Formula:
            risk_amount = balance * risk_pct
            loss_per_btc = entry_price * stop_loss_pct
            size = risk_amount / loss_per_btc

        Example:
            balance=$1000, risk_pct=2%, SL=1.5%, price=$60000
            risk_amount = $20
            loss_per_btc = $900
            size = 0.0222 BTC
        """
        risk_amount = self.balance * self.risk_pct
        loss_per_unit = entry_price * self.stop_loss_pct

        if loss_per_unit <= 0:
            return 0.0

        size = risk_amount / loss_per_unit

        # Safety cap: never risk more than 10% of balance worth
        max_size = (self.balance * 0.10) / entry_price
        return round(min(size, max_size), 6)

    def can_trade(self, daily_pnl: float) -> tuple[bool, str]:
        """
        Daily loss limit check.
        Returns (True, "") if OK to trade.
        Returns (False, reason) if limit hit.
        """
        max_daily_loss = self.balance * 0.05   # 5% daily loss limit
        if daily_pnl < -max_daily_loss:
            return False, f"Daily loss limit hit (${-daily_pnl:.2f} lost today)"
        return True, ""

def build_chart(

# ──────────────────────────────────────────────────────────────
# CHART BUILDER FUNCTION

def build_chart(
    df: pd.DataFrame,
    trades: list,
    show_ema: bool,
    show_rsi: bool,
    show_vol: bool
) -> go.Figure:
    """
    Build the full Plotly chart.
    Panels: Candlestick | EMA lines | BUY/SELL markers |
            Active position lines | RSI | Volume
    """

    # ── Empty state ─────────────────────────────────────────
    if df.empty:
        fig = go.Figure()
        fig.add_annotation(
            text="No data available — check API connection",
            xref="paper", yref="paper",
            x=0.5, y=0.5,
            showarrow=False,
            font=dict(size=16, color="#718096")
        )
        fig.update_layout(
            paper_bgcolor="#0e1117",
            plot_bgcolor="#0e1117",
            height=500
        )
        return fig

    # ── Build subplot grid ──────────────────────────────────
    row_count    = 1
    row_heights  = [0.7]
    specs        = [[{"type": "candlestick"}]]
    sub_titles   = ["BTC/USDT"]

    if show_rsi:
        row_count += 1
        row_heights.append(0.15)
        specs.append([{"type": "scatter"}])
        sub_titles.append("RSI (14)")

    if show_vol:
        row_count += 1
        row_heights.append(0.15)
        specs.append([{"type": "bar"}])
        sub_titles.append("Volume")

    fig = make_subplots(
        rows=row_count,
        cols=1,
        shared_xaxes=True,
        vertical_spacing=0.03,
        row_heights=row_heights,
        subplot_titles=sub_titles,
        specs=specs
    )

    # ── Candlestick ─────────────────────────────────────────
    fig.add_trace(
        go.Candlestick(
            x=df["timestamp"],
            open=df["open"],
            high=df["high"],
            low=df["low"],
            close=df["close"],
            name="BTC/USDT",
            increasing_line_color="#48bb78",
            decreasing_line_color="#fc8181",
            increasing_fillcolor="#48bb78",
            decreasing_fillcolor="#fc8181",
        ),
        row=1, col=1
    )

    # ── EMA Lines ───────────────────────────────────────────
    if show_ema and "ema9" in df.columns:
        fig.add_trace(
            go.Scatter(
                x=df["timestamp"],
                y=df["ema9"],
                name="EMA 9",
                line=dict(color="#63b3ed", width=1.5),
                hovertemplate="EMA9: $%{y:,.2f}<extra></extra>"
            ),
            row=1, col=1
        )
        fig.add_trace(
            go.Scatter(
                x=df["timestamp"],
                y=df["ema21"],
                name="EMA 21",
                line=dict(color="#f6e05e", width=1.5),
                hovertemplate="EMA21: $%{y:,.2f}<extra></extra>"
            ),
            row=1, col=1
        )
        if "ema50" in df.columns:
            fig.add_trace(
                go.Scatter(
                    x=df["timestamp"],
                    y=df["ema50"],
                    name="EMA 50",
                    line=dict(color="#b794f4", width=1.5, dash="dot"),
                    hovertemplate="EMA50: $%{y:,.2f}<extra></extra>"
                ),
                row=1, col=1
            )

    # ── BUY / SELL Markers ──────────────────────────────────
    # Each trade is snapped to the nearest candle timestamp
    # so markers sit exactly ON a candle — never floating

    if trades:
        buy_times   = []
        buy_prices  = []
        buy_texts   = []
        sell_times  = []
        sell_prices = []
        sell_texts  = []

        for trade in trades:
            entry_ts = pd.Timestamp(trade["entry_time"])
            exit_ts  = pd.Timestamp(trade["exit_time"])

            # Snap to nearest candle index
            entry_idx = int(abs(df["timestamp"] - entry_ts).argmin())
            exit_idx  = int(abs(df["timestamp"] - exit_ts).argmin())

            entry_row = df.iloc[entry_idx]
            exit_row  = df.iloc[exit_idx]

            # BUY marker — just below candle low
            buy_times.append(entry_row["timestamp"])
            buy_prices.append(entry_row["low"] * 0.9985)
            buy_texts.append(
                f"<b>BUY</b> @ ${trade['entry_price']:,.2f}<br>"
                f"Reason: {trade['reason']}<br>"
                f"Size: {trade['size']:.6f} BTC"
            )

            # SELL marker — just above candle high
            sell_times.append(exit_row["timestamp"])
            sell_prices.append(exit_row["high"] * 1.0015)
            pnl_str = (
                f"+${trade['pnl']:.4f}"
                if trade["pnl"] >= 0
                else f"-${abs(trade['pnl']):.4f}"
            )
            sell_texts.append(
                f"<b>SELL</b> @ ${trade['exit_price']:,.2f}<br>"
                f"Reason: {trade['reason']}<br>"
                f"PnL: {pnl_str}"
            )

        # BUY markers — green triangles up
        if buy_times:
            fig.add_trace(
                go.Scatter(
                    x=buy_times,
                    y=buy_prices,
                    mode="markers+text",
                    name="BUY",
                    marker=dict(
                        symbol="triangle-up",
                        size=14,
                        color="#48bb78",
                        line=dict(color="#276749", width=1)
                    ),
                    text=["▲"] * len(buy_times),
                    textposition="bottom center",
                    textfont=dict(size=10, color="#48bb78"),
                    hovertext=buy_texts,
                    hoverinfo="text",
                    showlegend=True
                ),
                row=1, col=1
            )

        # SELL markers — red triangles down
        if sell_times:
            fig.add_trace(
                go.Scatter(
                    x=sell_times,
                    y=sell_prices,
                    mode="markers+text",
                    name="SELL",
                    marker=dict(
                        symbol="triangle-down",
                        size=14,
                        color="#fc8181",
                        line=dict(color="#9b2c2c", width=1)
                    ),
                    text=["▼"] * len(sell_times),
                    textposition="top center",
                    textfont=dict(size=10, color="#fc8181"),
                    hovertext=sell_texts,
                    hoverinfo="text",
                    showlegend=True
                ),
                row=1, col=1
            )

    # ── Active Position Lines ────────────────────────────────
    if bot.position:
        entry_p = bot.position["entry_price"]
        tp_p    = bot.position["take_profit"]
        sl_p    = bot.position["stop_loss"]

        fig.add_hline(
            y=entry_p,
            line_dash="dash",
            line_color="#63b3ed",
            line_width=1,
            annotation_text=f"  Entry ${entry_p:,.2f}",
            annotation_font_color="#63b3ed",
            row=1, col=1
        )
        fig.add_hline(
            y=tp_p,
            line_dash="dot",
            line_color="#48bb78",
            line_width=1,
            annotation_text=f"  TP ${tp_p:,.2f}",
            annotation_font_color="#48bb78",
            row=1, col=1
        )
        fig.add_hline(
            y=sl_p,
            line_dash="dot",
            line_color="#fc8181",
            line_width=1,
            annotation_text=f"  SL ${sl_p:,.2f}",
            annotation_font_color="#fc8181",
            row=1, col=1
        )

    # ── RSI Panel ───────────────────────────────────────────
    rsi_row = None
    vol_row = None

    if show_rsi and "rsi" in df.columns:
        rsi_row = 2
        fig.add_trace(
            go.Scatter(
                x=df["timestamp"],
                y=df["rsi"],
                name="RSI",
                line=dict(color="#b794f4", width=1.5),
                hovertemplate="RSI: %{y:.1f}<extra></extra>"
            ),
            row=rsi_row, col=1
        )
        fig.add_hrect(
            y0=70, y1=100,
            fillcolor="#fc8181", opacity=0.08,
            layer="below", line_width=0,
            row=rsi_row, col=1
        )
        fig.add_hrect(
            y0=0, y1=30,
            fillcolor="#48bb78", opacity=0.08,
            layer="below", line_width=0,
            row=rsi_row, col=1
        )
        fig.add_hline(
            y=70, line_dash="dot",
            line_color="#fc8181", line_width=0.8,
            row=rsi_row, col=1
        )
        fig.add_hline(
            y=30, line_dash="dot",
            line_color="#48bb78", line_width=0.8,
            row=rsi_row, col=1
        )
        fig.update_yaxes(
            range=[0, 100],
            row=rsi_row, col=1,
            tickfont=dict(size=10)
        )

    # ── Volume Panel ────────────────────────────────────────
    if show_vol and "volume" in df.columns:
        vol_row = (rsi_row + 1) if rsi_row else 2

        vol_colors = [
            "#48bb78" if float(c) >= float(o) else "#fc8181"
            for c, o in zip(df["close"], df["open"])
        ]

        fig.add_trace(
            go.Bar(
                x=df["timestamp"],
                y=df["volume"],
                name="Volume",
                marker_color=vol_colors,
                opacity=0.7,
                hovertemplate="Vol: %{y:,.0f}<extra></extra>"
            ),
            row=vol_row, col=1
        )

    # ── Global Layout ────────────────────────────────────────
    fig.update_layout(
        height=700 if (show_rsi or show_vol) else 520,
        paper_bgcolor="#0e1117",
        plot_bgcolor="#131722",
        font=dict(color="#e2e8f0", family="Inter, sans-serif"),
        legend=dict(
            orientation="h",
            yanchor="bottom",
            y=1.02,
            xanchor="right",
            x=1,
            bgcolor="rgba(0,0,0,0)",
            font=dict(size=11)
        ),
        margin=dict(l=10, r=10, t=40, b=10),
        xaxis_rangeslider_visible=False,
        hovermode="x unified",
    )

    fig.update_xaxes(
        gridcolor="#1a1d27",
        showgrid=True,
        zeroline=False
    )
    fig.update_yaxes(
        gridcolor="#1a1d27",
        showgrid=True,
        zeroline=False
    )

    return fig

"""
strategy.py - Trading signal generation
Clean, readable, no side effects.
Input: enriched DataFrame (with indicators)
Output: {"action": "BUY"|"SELL"|"HOLD", "reason": str}
"""

import pandas as pd


def generate_signal(df: pd.DataFrame) -> dict:
    """
    EMA crossover + RSI filter strategy.

    BUY  when: EMA9 crosses above EMA21 AND RSI < 65 (not overbought)
    SELL when: EMA9 crosses below EMA21 AND RSI > 35 (not oversold)
    HOLD otherwise.
    """

    if len(df) < 3:
        return {"action": "HOLD", "reason": "Not enough data"}

    prev = df.iloc[-2]
    curr = df.iloc[-1]

    # Guard: check required columns exist
    required = ["ema9", "ema21", "rsi"]
    for col in required:
        if col not in df.columns or pd.isna(curr[col]):
            return {"action": "HOLD", "reason": f"Indicator '{col}' not ready"}

    ema9_crossed_up   = (prev["ema9"] < prev["ema21"]) and (curr["ema9"] > curr["ema21"])
    ema9_crossed_down = (prev["ema9"] > prev["ema21"]) and (curr["ema9"] < curr["ema21"])

    rsi = curr["rsi"]
    rsi_not_overbought = rsi < 65
    rsi_not_oversold   = rsi > 35

    # ── BUY signal ──────────────────────────────────────────
    if ema9_crossed_up and rsi_not_overbought:
        return {
            "action": "BUY",
            "reason": f"EMA9 crossed above EMA21 | RSI={rsi:.1f}"
        }

    # ── SELL signal ─────────────────────────────────────────
    if ema9_crossed_down and rsi_not_oversold:
        return {
            "action": "SELL",
            "reason": f"EMA9 crossed below EMA21 | RSI={rsi:.1f}"
        }

    return {"action": "HOLD", "reason": "No crossover signal"}


def get_trend(df: pd.DataFrame) -> str:
    """
    Simple trend label for display.
    Based on price vs EMA50.
    Returns a readable string shown in the dashboard metric.
    """

    # Guard: empty dataframe
    if df is None or df.empty:
        return "Unknown"

    # Guard: ema50 column missing
    if "ema50" not in df.columns:
        return "Unknown"

    price = df["close"].iloc[-1]
    ema50 = df["ema50"].iloc[-1]

    # Guard: NaN value
    if pd.isna(ema50) or pd.isna(price):
        return "Unknown"

    if price > ema50 * 1.005:
        return "Bullish"
    elif price < ema50 * 0.995:
        return "Bearish"
    else:
        return "Neutral"

"""
risk.py - Risk management
Handles: position sizing, daily loss limits
"""


class RiskManager:
    """
    Manages position sizing and risk rules.
    Used by TradingBot to calculate trade sizes and limits.
    """

    def __init__(
        self,
        balance: float       = 1000.0,
        risk_pct: float      = 0.02,
        stop_loss_pct: float = 0.015,
        take_profit_pct: float = 0.03
    ):
        self.balance         = balance
        self.risk_pct        = risk_pct
        self.stop_loss_pct   = stop_loss_pct
        self.take_profit_pct = take_profit_pct

    def position_size(self, entry_price: float) -> float:
        """
        Calculate position size in BTC.

        Formula:
            risk_amount  = balance * risk_pct
            loss_per_btc = entry_price * stop_loss_pct
            size         = risk_amount / loss_per_btc

        Example:
            balance      = $1000
            risk_pct     = 2%      → risk_amount  = $20
            stop_loss    = 1.5%    → loss_per_btc = $900 (at $60000)
            size         = 0.0222 BTC

        Safety cap:
            Never use more than 10% of balance worth of BTC.
        """
        if entry_price <= 0:
            return 0.0

        risk_amount   = self.balance * self.risk_pct
        loss_per_unit = entry_price * self.stop_loss_pct

        if loss_per_unit <= 0:
            return 0.0

        size = risk_amount / loss_per_unit

        # Safety cap — never exceed 10% of balance in one trade
        max_size = (self.balance * 0.10) / entry_price
        return round(min(size, max_size), 6)

    def can_trade(self, daily_pnl: float) -> tuple:
        """
        Daily loss limit check.

        Returns:
            (True,  "")       → OK to trade
            (False, reason)   → limit hit, stop trading
        """
        max_daily_loss = self.balance * 0.05    # 5% daily loss limit

        if daily_pnl < -max_daily_loss:
            return (
                False,
                f"Daily loss limit hit (${abs(daily_pnl):.2f} lost today, "
                f"max allowed: ${max_daily_loss:.2f})"
            )

        return (True, "")

    def get_tp_sl(self, entry_price: float) -> dict:
        """
        Calculate take profit and stop loss prices from entry.

        Returns:
            {
                "take_profit": float,
                "stop_loss":   float
            }
        """
        tp = round(entry_price * (1 + self.take_profit_pct), 2)
        sl = round(entry_price * (1 - self.stop_loss_pct),   2)
        return {"take_profit": tp, "stop_loss": sl}

    def update_balance(self, new_balance: float):
        """Update balance after a trade closes."""
        self.balance = round(new_balance, 4)

    def summary(self) -> dict:
        """Return current risk settings as a dict for display."""
        return {
            "balance":         self.balance,
            "risk_pct":        f"{self.risk_pct * 100:.1f}%",
            "stop_loss_pct":   f"{self.stop_loss_pct * 100:.1f}%",
            "take_profit_pct": f"{self.take_profit_pct * 100:.1f}%",
        }

        ts    = self.df["timestamp"].iloc[-1]
        if self.position is not None:
            self.log(
                f"Manual BUY rejected — already in position "
                f"@ ${self.position['entry_price']:,.2f}",
                "INFO"
            )
            return
        self._open_position(price, ts, "Manual BUY")

    def manual_sell(self):
        if self.df.empty:
            self.log("No data for manual SELL", "ERROR")
            return
        price = float(self.df["close"].iloc[-1])
        ts    = self.df["timestamp"].iloc[-1]
        if self.position is None:
            self.log("Manual SELL rejected — no open position", "INFO")
            return
        self._close_position(price, ts, "Manual SELL")

    # ──────────────────────────────────────────────────────
    # BOT LOOP — fetches every 5 seconds
    # ──────────────────────────────────────────────────────

    def _bot_loop(self):
        self.log("Bot loop started", "SYSTEM")

        while self.running:
            try:
                # 1. Fetch OHLC
                df = self.fetch_ohlc()

                if df.empty:
                    self.log("No data — retrying in 10s", "INFO")
                    time.sleep(10)
                    continue

                # 2. Add indicators
                df = self._add_indicators(df)

                # 3. Fetch live price
                live_price = self.fetch_price()
                if live_price:
                    self.last_price = live_price

                # 4. Atomically update shared dataframe
                with self._lock:
                    self.df            = df.copy()
                    self.last_fetch_ts = datetime.now()

                # 5. Check TP/SL
                if self.position and live_price:
                    self._check_tp_sl(live_price, datetime.now())

                # 6. Auto signal
                if self.auto_mode:
                    signal = generate_signal(df)
                    self._execute_signal(signal, df)

            except Exception as e:
                self.log(f"Loop error: {e}", "ERROR")

            # Sleep 5 seconds between fetches
            # Check running flag every 0.5s so stop() is instant
            for _ in range(10):
                if not self.running:
                    break
                time.sleep(0.5)

        self.log("Bot loop stopped", "SYSTEM")

    def _execute_signal(self, signal: dict, df: pd.DataFrame):
        action = signal.get("action")
        reason = signal.get("reason", "")
        latest = df.iloc[-1]
        price  = latest["close"]
        ts     = latest["timestamp"]

        if action == "BUY" and self.position is None:
            self._open_position(price, ts, f"Auto: {reason}")
        elif action == "SELL" and self.position is not None:
            self._close_position(price, ts, f"Auto: {reason}")

    # ──────────────────────────────────────────────────────
    # START / STOP
    # ──────────────────────────────────────────────────────

    def start(self):
        """
        Start the bot loop.
        Checks both self.running AND thread.is_alive()
        to guarantee no duplicate threads ever spawn.
        """
        if self.running:
            self.log("Already running", "INFO")
            return

        if self._thread is not None and self._thread.is_alive():
            self.log("Thread already alive", "INFO")
            return

        self.running = True
        self._thread = threading.Thread(
            target=self._bot_loop,
            daemon=True,
            name="TradingBotThread"
        )
        self._thread.start()
        self.log("Bot started", "SYSTEM")

    def stop(self):
        """Stop the bot loop gracefully."""
        if not self.running:
            self.log("Already stopped", "INFO")
            return
        self.running = False
        self.log("Bot stopped", "SYSTEM")

    def toggle_auto(self):
        self.auto_mode = not self.auto_mode
        state = "ON" if self.auto_mode else "OFF"
        self.log(f"Auto trading: {state}", "SYSTEM")

    def is_thread_alive(self) -> bool:
        """Safe check — used by dashboard to verify thread health."""
        return self._thread is not None and self._thread.is_alive()

    # ──────────────────────────────────────────────────────
    # STATS
    # ──────────────────────────────────────────────────────

    def get_stats(self) -> dict:
        total_trades = len(self.trades)
        winning      = sum(1 for t in self.trades if t["pnl"] > 0)
        losing       = total_trades - winning
        total_pnl    = sum(t["pnl"] for t in self.trades)
        win_rate     = (winning / total_trades * 100) if total_trades > 0 else 0.0

        return {
            "balance":      self.balance,
            "pnl":          total_pnl,
            "total_trades": total_trades,
            "winning":      winning,
            "losing":       losing,
            "win_rate":     win_rate,
            "in_position":  self.position is not None,
            "position":     self.position
        }
